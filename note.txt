## Example Full Code (main.jsx)

import React from "react";
import ReactDOM from "react-dom/client";
import { createBrowserRouter, RouterProvider } from "react-router-dom";

const router = createBrowserRouter([
  {
    path: "/",
    element: <div>Welcome to React Router!</div>,
  },
]);

ReactDOM.createRoot(document.getElementById("root")).render(
  <React.StrictMode>
    default (<App></App>)
    <RouterProvider router={router} />
  </React.StrictMode>
);

## Nested Route Basics Concepts.
--------------------------------

** When we want to routes inside other routes(like a layout with a header/sidebar), we use nested routes.
1. Create Layout Component (shared UI)
---------------------------------
function Layout() {
  return (
    <div>
      <h1>My App Layout</h1>
      <Outlet></Outlet>   //  Renders child routes here(page eh kon jayga theke amra baki child gula dekhte cai).
    </div>
  );
}
